# 0037. 解数独

## 题目
编写一个程序，通过填充空格来解决数独问题。

数独的解法需 遵循如下规则：

数字 1-9 在每一行只能出现一次。
数字 1-9 在每一列只能出现一次。
数字 1-9 在每一个以粗实线分隔的 3x3 宫内只能出现一次。（请参考示例图）
数独部分空格内已填入了数字，空白格用 '.' 表示。

class L0037SolveSudoku {

    public void solveSudoku(char[][] board) {
        sudokuBacktrack(board);
    }

    private boolean sudokuBacktrack(char[][] b) {
        for (int i = 0; i < b.length; i++) {
            for (int j = 0; j < b[0].length; j++) {
                if (b[i][j] != '.') {
                    continue;
                }
                for (char v = '1'; v <= '9'; v++) {
                    if (sudokuValid(b, i, j, v)) { // 尝试放置字符v
                        b[i][j] = v;
                        if (sudokuBacktrack(b)) { // 以此基础的后续尝试成功后返回
                            return true;
                        }
                        b[i][j] = '.';
                    }
                }
                return false; // 当前尝试所剩字符也不满足
            }
        }
        return true;
    }

    private boolean sudokuValid(char[][] b, int i, int j, char v) {
        for (int k = 0; k < b[i].length; k++) {
            if (b[i][k] == v) {
                return false;
            }
        }
        for (int k = 0; k < b.length; k++) {
            if (b[0][j] == v) {
                return false;
            }
        }
        int r = (i / 3) * 3, c = (j / 3) * 3;
        for (int k = 0; k < 3; k++) {
            for (int m = 0; m < 3; m++) {
                if (b[r + k][c + m] == v) {
                    return false;
                }
            }
        }
        return true;
    }

}
